// Efecto Parallax (ajustado para mayor control)
document.addEventListener('scroll', function() {
    const parallaxSections = document.querySelectorAll('.parallax');
    parallaxSections.forEach(section => {
        const speed = parseFloat(section.dataset.parallaxSpeed) || 0.3;
        const offset = window.pageYOffset * speed;
        section.style.backgroundPositionY = `${offset}px`;
    });
});

// Efecto de aparición gradual al hacer scroll (Intersection Observer API)
const sections = document.querySelectorAll('.section');
const observer = new IntersectionObserver((entries, observer) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            entry.target.classList.add('visible');
            observer.unobserve(entry.target);
        }
    });
}, {
    threshold: 0.15
});
sections.forEach(section => {
    observer.observe(section);
});

});

// Indicador de navegación activa
const navLinks = document.querySelectorAll('nav a');
function updateActiveNavLink() {
    const currentScrollPos = window.pageYOffset;
    navLinks.forEach(link => {
        const targetId = link.getAttribute('href').substring(1);
        const targetSection = document.getElementById(targetId);
        if (targetSection &&
            targetSection.offsetTop <= currentScrollPos + window.innerHeight / 2 &&
            targetSection.offsetTop + targetSection.offsetHeight > currentScrollPos + window.innerHeight / 2) {
            link.classList.add('active');
        } else {
            link.classList.remove('active');
        }
    });
}
window.addEventListener('scroll', updateActiveNavLink);
window.addEventListener('load', updateActiveNavLink);

const carrusel = document.querySelector('.carousel-container');
    const slide = document.querySelector('.carousel-slide');
    const elementos = document.querySelectorAll('.tech-item');
    const botonAnterior = document.querySelector('.prev-control');
    const botonSiguiente = document.querySelector('.next-control');
    
    let indiceActual = 0;
    const anchoElemento = elementos[0].offsetWidth; // Ancho de un elemento
    const totalElementos = elementos.length;
    
    function mostrarElemento(indice) {
        if (indice < 0) {
            indiceActual = totalElementos - 1;
        } else if (indice >= totalElementos) {
            indiceActual = 0;
        } else {
            indiceActual = indice;
        }
    
        slide.style.transform = `translateX(${-indiceActual * anchoElemento}px)`;
    }
    
    botonSiguiente.addEventListener('click', () => {
        mostrarElemento(indiceActual + 1);
    });
    
    botonAnterior.addEventListener('click', () => {
        mostrarElemento(indiceActual - 1);
    });
    
    mostrarElemento(0); // Mostrar el primer elemento al inicio
